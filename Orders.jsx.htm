import React from 'react'
import { Box, Heading, Table, Thead, Tbody, Tr, Th, Td, Badge } from '@chakra-ui/react'

function Orders() {
const orders = [
{ id: 1, customer: 'John Doe', total: 59.97, status: 'Completed' },
{ id: 2, customer: 'Jane Smith', total: 89.98, status: 'Processing' },
{ id: 3, customer: 'Bob Johnson', total: 39.99, status: 'Shipped' },
]

return (
<Box>
<Heading mb={4}>Orders</Heading>
<Table variant="simple">
<Thead>
<Tr>
<Th>ID</Th>
<Th>Customer</Th>
<Th>Total</Th>
<Th>Status</Th>
</Tr>
</Thead>
<Tbody>
{orders.map((order) => (
<Tr key={order.id}>
<Td>{order.id}</Td>
<Td>{order.customer}</Td>
<Td>${order.total}</Td>
<Td>
<Badge colorScheme={order.status === 'Completed' ? 'green' : 'orange'}>
{order.status}
</Badge>
</Td>
</Tr>
))}
</Tbody>
</Table>
</Box>
)
}

export default Orders
